### My SQL Query
```sql
WITH TweetStats AS (
  SELECT user_id, 
  COUNT(*) AS tweet_count 
  FROM tweets 
  GROUP BY user_id
),
UserStats AS (
  SELECT user_id, user_name FROM users
)
SELECT 
  us.user_name,
  ts.tweet_count
  FROM TweetStats ts 
  JOIN UserStats us ON ts.user_id = us.user_id
ORDER BY tweet_count DESC;


```

#### Tweets & Avg Length by User

SELECT u.user_name, COUNT(t.tweet_id) as tweet_count, ROUND(AVG(LENGTH(t.text)),1) as avg_text_len FROM tweets t JOIN users u ON t.user_id = u.user_id GROUP BY u.user_name ORDER BY tweet_count DESC LIMIT 10
